// Token pair scoring algorithm class PairSelector { scorePair(assetPair, timeframe = '30d') { const metrics = this.calculateMetrics(assetPair, timeframe); const score = { feeGeneration: metrics.avgFeeRate * 0.40, // 40% weight riskAdjusted: metrics.sharpeRatio * 0.30, // 30% weight liquidityDepth: metrics.liquidity * 0.15, // 15% weight volatilityStability: metrics.volStability * 0.10, // 10% weight marketGrowth: metrics.growthPotential * 0.05 // 5% weight }; return Object.values(score).reduce((a, b) => a + b, 0); } async recommendOptimalPairs(portfolioSize) { const allPairs = await this.scanMarket(); const scoredPairs = allPairs.map(pair => ({ ...pair, score: this.scorePair(pair) })); return this.optimizePortfolio(scoredPairs, portfolioSize); } }
